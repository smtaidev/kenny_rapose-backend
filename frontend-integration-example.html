// Replace with your actual API endpoint and package IDs const API_BASE_URL = 'http://localhost:5000/api/v1'; const PACKAGE_IDS = { starter: 'your-starter-package-id', pro: 'your-pro-package-id', enterprise: 'your-enterprise-package-id' }; async function purchaseCredits(packageType) { try { // Get user token from your auth system const token = localStorage.getItem('accessToken'); if (!token) { alert('Please log in to purchase credits'); return; } const response = await fetch(`${API_BASE_URL}/payments/create-checkout-session`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` }, body: JSON.stringify({ packageId: PACKAGE_IDS[packageType], successUrl: `${window.location.origin}/payment/success`, cancelUrl: `${window.location.origin}/payment/cancel` }) }); if (!response.ok) { throw new Error('Failed to create checkout session'); } const { sessionUrl } = await response.json(); // Redirect to Stripe Checkout window.location.href = sessionUrl; } catch (error) { console.error('Payment error:', error); showPaymentStatus('error', 'Failed to create checkout session. Please try again.'); } } function showPaymentStatus(type, message) { const statusDiv = document.getElementById('payment-status'); statusDiv.className = `payment-status ${type}`; statusDiv.textContent = message; statusDiv.style.display = 'block'; setTimeout(() => { statusDiv.style.display = 'none'; }, 5000); }
